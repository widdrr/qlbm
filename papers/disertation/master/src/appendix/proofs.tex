\section{Full Schaffer form derivation}
\label{app:Schaffer}
\begin{align*}
    U &= \begin{pmatrix} A & C \\ B & D \end{pmatrix} \\
    U^\dagger &= \begin{pmatrix} A^\dagger & B^\dagger \\ C^\dagger & D^\dagger \end{pmatrix} \\
    \intertext{Since $U$ must be unitary, it follows that}
    U^\dagger U &= \begin{pmatrix} A^\dagger A + B^\dagger B & A^\dagger C + B^\dagger D \\ 
    C^\dagger A + D^\dagger B & C^\dagger C + D^\dagger D \end{pmatrix} = \begin{pmatrix} I & 0 \\ 0 & I \end{pmatrix} \\
    \intertext{From the top-left block, we derive $B$:}
    B^\dagger B &= I - A^\dagger A \implies B = \sqrt{I - A^\dagger A} \\
    \intertext{To find $C$, we evaluate the other unitarity condition, $U U^\dagger = I$:}
    U U^\dagger &= \begin{pmatrix} A A^\dagger + C C^\dagger & A B^\dagger + C D^\dagger \\ 
    B A^\dagger + D C^\dagger & B B^\dagger + D D^\dagger \end{pmatrix} = \begin{pmatrix} I & 0 \\ 0 & I \end{pmatrix} \\
    \intertext{From the top-left block of this result, we isolate $C$:}
    C C^\dagger &= I - A A^\dagger \implies C = \sqrt{I - A A^\dagger} \\
    \intertext{Note that $B$ and $C$ are both Hermitian. Lastly, to solve for $D$, we use the off-diagonal term from $U^\dagger U$:}
    A^\dagger C + B^\dagger D &= 0 \\
    A^\dagger \sqrt{I - A A^\dagger} + \sqrt{I - A^\dagger A} \, D &= 0 \\
    \intertext{It can be proven via the power expansion of $f(x) = \sqrt{1-x}$ and associativity that $A^\dagger \sqrt{I - A A^\dagger} = \sqrt{I - A^\dagger A}A^\dagger$}
    \sqrt{I - A^\dagger A} A^\dagger + \sqrt{I - A^\dagger A} \, D &= 0 \implies D = -A^\dagger \\
    \intertext{Yielding the Schaffer form for the Sz.-Nagy dilation}
    U &= \begin{pmatrix} A & \sqrt{I - A A^\dagger} \\ \sqrt{I - A^\dagger A} & -A^\dagger \end{pmatrix}
\end{align*}

\section{Action of $P^\dagger$ for the 2-unitary example}
\label{app:inverse_prepare}

$P$ is defined by its action on the $\ket{0}$ state:
\begin{equation*}
    P\ket{0} = \sqrt{\frac{\alpha_0}{\lambda}}\ket{0} + \sqrt{\frac{\alpha_1}{\lambda}}\ket{1}
\end{equation*}

Typically $\alpha_j$ coefficients are all real, positive numbers. Even if that is not the case initially, it can be enforced by absorbing any phase into the unitary itself:
\begin{equation*}
    \alpha_j U_j = (re^{i\theta})U_j = r(e^{i\theta}U_j) = rU_j' = |\alpha_j|U'_j
\end{equation*}

To determine how $P^\dagger$ acts, we shall inspect the matrix form of $P$. 
\begin{align*}
    P = \begin{pmatrix} \sqrt{\frac{\alpha_0}{\lambda}} & c_0 \\ \sqrt{\frac{\alpha_1}{\lambda}} & c_1 \end{pmatrix} \\
    \intertext{with $c_0$ and $c_1$ being two constants such that the $P$ is unitary. $P\dagger$ is thus:}
    P^\dagger = \begin{pmatrix} \sqrt{\frac{\alpha_0}{\lambda}} & \sqrt{\frac{\alpha_1}{\lambda}} \\ c_{0}^* & c_{1}^* \end{pmatrix}
\end{align*}

Applying $P^\dagger$ to $\ket{0}$ and $\ket{1}$:
\begin{align*}
    P^\dagger\ket{0} = \begin{pmatrix} \sqrt{\frac{\alpha_0}{\lambda}} & \sqrt{\frac{\alpha_1}{\lambda}} \\ c_{0}^* & c_{1}^* \end{pmatrix} \begin{pmatrix} 1 \\ 0 \end{pmatrix} = \begin{pmatrix} \sqrt{\frac{\alpha_0}{\lambda}} \\ c_{0}^* \end{pmatrix} = \sqrt{\frac{\alpha_0}{\lambda}}\ket{0} + c_{0}^*\ket{1} \\
    P^\dagger\ket{1} = \begin{pmatrix} \sqrt{\frac{\alpha_0}{\lambda}} & \sqrt{\frac{\alpha_1}{\lambda}} \\ c_{0}^* & c_{1}^* \end{pmatrix} \begin{pmatrix} 0 \\ 1 \end{pmatrix} = \begin{pmatrix} \sqrt{\frac{\alpha_1}{\lambda}} \\ c_{1}^* \end{pmatrix} = \sqrt{\frac{\alpha_1}{\lambda}}\ket{0} + c_{1}^*\ket{1}
\end{align*}

Substituting these explicit mappings directly:
\begin{align*}
    (P^\dagger \otimes I) &\left( \sqrt{\frac{\alpha_0}{\lambda}}\ket{0}U_0\ket{\psi} + \sqrt{\frac{\alpha_1}{\lambda}}\ket{1}U_1\ket{\psi} \right) \nonumber \\
    &= \sqrt{\frac{\alpha_0}{\lambda}} \left( \sqrt{\frac{\alpha_0}{\lambda}}\ket{0} + c_0^*\ket{1} \right) U_0\ket{\psi} + \sqrt{\frac{\alpha_1}{\lambda}} \left( \sqrt{\frac{\alpha_1}{\lambda}}\ket{0} + c_1^*\ket{1} \right) U_1\ket{\psi} \nonumber \\
    \intertext{Factoring out the $\ket{0}$ ancilla state reveals the recovered matrix addition:}
    &= \ket{0} \left( \frac{\alpha_0}{\lambda} U_0 + \frac{\alpha_1}{\lambda} U_1 \right) \ket{\psi} + \ket{1}\ket{\Phi^\perp} \nonumber \\
    &= \frac{1}{\lambda} \ket{0} A \ket{\psi} + \ket{1}\ket{\Phi^\perp}
\end{align*}

\section{Proof of Collision Operator Decomposition}
\label{app:collision_lcu}

Given a diagonal, real, and contractive collision matrix $M$, it can be decomposed as:

\begin{align*}
    M &= \frac{1}{2}(U_0 + U_1) \\
    \intertext{where the unitary operators are defined as:}
    U_0 &= M + i\sqrt{I - M^2} \\
    U_1 &= M - i\sqrt{I - M^2}
\end{align*}
This particular construction is defined, albeit in a different form, in Low and Chang's Qubitization Framework \cite{low2019hamiltonian}.

First of all, to prove this construction recovers $M$, simple matrix algebra is enough

\begin{align*}
    \frac{1}{2}(U_0 + U_1) &= \frac{1}{2} \left( \left[M + i\sqrt{I - M^2}\right] + \left[M - i\sqrt{I - M^2}\right] \right) \\
    &= \frac{1}{2} \left( 2M + i\sqrt{I - M^2} - i\sqrt{I - M^2} \right) \\
    &= M
\end{align*}

Next we need to prove that $U_0$ and $U_1$ are indeed unitary.
We'll focus on $U_0$ and show that $U_0^\dagger U_0 = U_0 U_0^\dagger = I$.

Because $M$ is both real-valued and diagonal, it follows that $M$ is Hermitian ($M = M^\dagger$).
Additionally it is also contractive ($\|M\|_2 = \sigma_{\max}(M) \le 1$), meaning $\sqrt{I-M^2}$ is also Hermitian.
We then have:
\begin{align*}
    U_0^\dagger &= (M + i\sqrt{I-M^2})^\dagger \\
    &= M^\dagger - i(\sqrt{I - M^2})^\dagger \\
    &= M - i\sqrt{I - M^2}
\end{align*}

Calculating the product:

\begin{align*}
    U_0^\dagger U_0 &= \left( M - i\sqrt{I - M^2} \right) \left( M + i\sqrt{I - M^2} \right) \\
    &= M^2 + iM\sqrt{I - M^2} - i\sqrt{I - M^2}M + \left( \sqrt{I - M^2} \right)^2
\end{align*}

Since both $M$ and $\sqrt{I-M^2}$ are diagonal, they commute, causing the cross-terms to vanish:

\begin{align*}
    U_0^\dagger U_0 &= M^2 + (I - M^2) \\
    &= I
\end{align*}

The exact same logic is used to show that $U_0 U_{0}^\dagger = I$.
The entire process is repeated for $U_1$.

\section{Variance Stabilization in Amplitude-Encoded Readout}
\label{app:variance_stabilization}


The L2-normalized link distributions are encoded as amplitudes of the quantum state. 
After an iteration, the state $\ket{x,i}$ will have amplitude of absolut value $f_i(x)$, where $i$ is the link index and $x$ is the site.
As per the Born rule, the probability of measuring $\ket{x,i}$ will be $f_i(x)^2 = p_i(x)$. 
All notation thereafter is implied to refer to link $i$ and site $x$, even if omitted. We will also ignore the ancilla qubit for simplicity.

Denote $K ~ B(S,p)$ the random variable determining the number of times we measure our state out of $S$ trials.
Let $\hat{p} = K/S$ be our statistical estimator for $p$. From standard binomial statistics we have:
\begin{align*}
    E[K] = Sp &\implies E[\hat{p}] = p \\
    \text{Var}[K] = Sp(1-p) &\implies \text{Var}(\hat{p}) = \frac{p(1 - p)}{S}
\end{align*}

However we need to recover the post-iteration $f = \sqrt{p}$, meaning we will be trying to determine $\hat{f} = \sqrt{\hat{p}}$.
To determine the variance we can apply the first-order Delta method \cite{casella2002statistical} which relies on the Taylor series expansion of the applied function around its mean $\mu$:
\begin{equation}
    \text{Var}(f(X)) \approx \left[ f'(\mu) \right]^2 \text{Var}(X)
\end{equation}

Applying this for $\hat{f} = \sqrt{\hat{p}}$ we obtain:

\begin{align*}
      \text{Var}\left(\sqrt{\hat{p}}\right) &\approx \left( \frac{1}{2\sqrt{p}} \right)^2 \left( \frac{p(1 - p)}{S} \right) \nonumber \\
    &\approx \left( \frac{1}{4p} \right) \left( \frac{p(1 - p)}{S} \right) \nonumber \\
    &\approx \frac{1 - p}{4S}
\end{align*}

In the context of the fluid simulation with $M$ lattice sites and $N$ discrete velocities, the probability must be split across the $M \times N$ states.
While the actual distribution of the fluid matters, in general as the scope of the simulation increases, the probability of measuring any specific state decreases $(p\ll1)$.
Under these circumstances, we can simplify the variance to:

\begin{equation}
    \text{Var}(\hat{f}) \approx \frac{1}{4S}
\end{equation}

This result prooves that the statistical variance of an approximated amplitude only depends on the total shot count.
Defining $\rho = \sum_{i=1}^{N}f_i$ to be the real density at the site, we obtained the estimator $\hat{\rho} = \sum_{i=0}^{N}\hat{f}_i$, with variance:
\begin{equation*}
    \text{Var}(\hat{\rho}) = \sum_{i=1}^N \text{Var}(\hat{f}_{i}) \approx \sum_{i=1}^N \frac{1}{4S} = \frac{N}{4S}
\end{equation*}

By definition, the Root Mean Square Error (RMSE) of the entire lattice density will be:
\begin{equation}
    \text{RMSE} = \sqrt{ \frac{1}{M} \sum_{x=1}^{M} E\left[ (\hat{\rho}_x - \rho_x)^2 \right] } = \sqrt{ \frac{1}{M} \sum_{s=x}^{M} \text{Var}(\hat{\rho}_x) }
\end{equation}

Applying our value for $\text{Var}(\hat{\rho}_x)$ we obtain:

\begin{equation*}
    \text{RMSE} \approx \sqrt{ \frac{1}{M} \sum_{x=1}^{M} \left( \frac{N}{4S} \right) } = \sqrt{ \frac{1}{M} \left( M \cdot \frac{N}{4S} \right) } = \frac{\sqrt{N}}{2\sqrt{S}}
\end{equation*}

Since the lattice size naturally vanishes from the RMSE and because each individual lattice site variance does not depend on it, neither does the final expression.
This result demonstrates that the error only depends on the total shot count and the number of link velocities.
The $N$ values used in our experiment (3 vs 5) are too insignificant compared to the shot count to be able to numerically see a difference.
